    <script>
        // ================= CONFIG =================
        const TARGET_URL = "https://snmbct.org";
        const AUTO_PILOT_START = 95;
        const AUDIO_START_TIME = 12; 
        
        // ================= STATE =================
        let charge = 0;
        let isComplete = false;
        let isAutoMode = false;
        let decayInterval;
        let audioStarted = false;
        let actualVolume = 0; 

        const btn = document.getElementById('powerBtn');
        const fill = document.getElementById('btnFill');
        const text = document.getElementById('chargeDisplay');
        const phaseText = document.getElementById('phaseIndicator');
        const meters = [document.getElementById('meterLeft'), document.getElementById('meterRight')];
        const ui = document.getElementById('uiContainer');
        const curtain = document.getElementById('launch-curtain');
        const meterWrappers = [document.getElementById('meterLeftWrapper'), document.getElementById('meterRightWrapper')];
        const bgm = document.getElementById('launchAudio');
        const countdownEl = document.getElementById('countdown-overlay');
        const graphicsEl = document.getElementById('countdown-graphics');
        const rocketWrapper = document.getElementById('rocket-wrapper');
        const rocketVideo = document.getElementById('rocket-video');

        // ================= AUDIO ENGINE =================
        function audioLoop() {
            if(isComplete) return; 
            let targetVolume = 0;
            if (charge > 0) { targetVolume = 0.6 + ((charge / 100) * 0.4); }
            actualVolume += (targetVolume - actualVolume) * 0.1;
            if(actualVolume < 0.001) actualVolume = 0;
            if(actualVolume > 1) actualVolume = 1;
            bgm.volume = actualVolume;
            requestAnimationFrame(audioLoop);
        }

        function triggerAudioStart() {
            if (!audioStarted) {
                audioStarted = true;
                bgm.currentTime = AUDIO_START_TIME;
                bgm.play().catch(e => console.log("Browser blocked audio"));
                audioLoop(); 
            }
        }

        // ================= GAME LOGIC =================
        function getStage(c) {
            if (c < 50) return { add: 20, sub: 10, name: "PHASE 1: TURBO" };
            return { add: 10, sub: 5, name: "PHASE 2: STABILIZE" };
        }

        function clickHandler(e) {
            if(e.cancelable) e.preventDefault();
            if(isComplete || isAutoMode) return;
            triggerAudioStart();
            const stage = getStage(charge);
            charge += stage.add;
            btn.style.transform = "scale(0.98)";
            setTimeout(() => btn.style.transform = "scale(1)", 50);
            if(charge >= AUTO_PILOT_START) { engageAutoPilot(); } else { updateVisuals(); }
        }

        function engageAutoPilot() {
            if(isAutoMode) return;
            isAutoMode = true;
            clearInterval(decayInterval); 
            charge = 95;
            phaseText.innerText = "PHASE 3: INJECTION";
            btn.classList.add('locked');
            function loop() {
                charge += 0.5; 
                if(charge >= 100) {
                    charge = 100;
                    updateVisuals();
                    triggerSuccess();
                } else {
                    updateVisuals();
                    requestAnimationFrame(loop);
                }
            }
            requestAnimationFrame(loop);
        }

        function startDecay() {
            decayInterval = setInterval(() => {
                if(isComplete || isAutoMode) return;
                const stage = getStage(charge);
                if(charge > 0) charge -= stage.sub;
                if(charge < 0) charge = 0;
                updateVisuals();
            }, 100);
        }

        function updateVisuals() {
            let show = Math.min(charge, 100);
            const stage = isAutoMode ? { name: "PHASE 3: INJECTION" } : getStage(charge);
            fill.style.width = show + "%";
            meters.forEach(m => m.style.height = show + "%");
            text.innerText = Math.floor(show);
            phaseText.innerText = stage.name;
        }

        // ================= CONFETTI =================
        const canvas = document.getElementById('confetti-canvas');
        const ctx = canvas.getContext('2d');
        let particles = [];
        function resizeCanvas() { canvas.width = window.innerWidth; canvas.height = window.innerHeight; }
        window.addEventListener('resize', resizeCanvas); resizeCanvas();
        function createConfetti() {
            const colors = ['#d4af37', '#ff9900', '#ffffff', '#ff5500'];
            function spawnExplosion(xOrigin, yOrigin) {
                for(let i=0; i<300; i++) {
                    particles.push({
                        x: xOrigin, y: yOrigin, vx: (Math.random() - 0.5) * 50, vy: (Math.random() - 1.2) * 40, 
                        size: Math.random() * 12 + 4, color: colors[Math.floor(Math.random() * colors.length)],
                        life: 200, drag: 0.92 
                    });
                }
            }
            spawnExplosion(0, window.innerHeight); spawnExplosion(window.innerWidth, window.innerHeight); animateConfetti();
        }
        function animateConfetti() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            particles.forEach((p, index) => {
                p.x += p.vx; p.y += p.vy; p.vx *= p.drag; p.vy *= p.drag; p.vy += 0.5; p.life--;
                ctx.fillStyle = p.color; ctx.globalAlpha = Math.max(0, p.life / 100); ctx.fillRect(p.x, p.y, p.size, p.size);
                if(p.life <= 0) particles.splice(index, 1);
            });
            if(particles.length > 0) requestAnimationFrame(animateConfetti);
        }

        // ================= SUCCESS SEQUENCE =================
        function triggerSuccess() {
            if(isComplete) return;
            isComplete = true;

            ui.classList.add('hidden');
            meterWrappers.forEach(w => w.classList.add('hidden'));

            graphicsEl.classList.add('active');
            countdownEl.classList.add('active');
            countdownEl.classList.add('pulsing');
            
            let timeLeft = 10;
            countdownEl.innerText = timeLeft;
            const countdownTimer = setInterval(() => {
                timeLeft--;
                if(timeLeft > 0) {
                    countdownEl.innerText = timeLeft;
                } else {
                    clearInterval(countdownTimer);
                    executeLaunch();
                }
            }, 1000);
        }

        function executeLaunch() {
            countdownEl.classList.remove('pulsing');
            countdownEl.style.fontSize = "4rem"; 
            countdownEl.innerText = "LAUNCHING...";
            graphicsEl.style.opacity = 0; 

            rocketVideo.play();
            
            setTimeout(() => {
                rocketWrapper.classList.add('launching');
                curtain.classList.add('launching');
            }, 50);

            setTimeout(createConfetti, 1500);

            const fadeDuration = 4000; 
            const startVol = bgm.volume;
            const step = startVol / (fadeDuration / 50);
            let fadeOut = setInterval(() => {
                if (bgm.volume >= step) { bgm.volume -= step; } 
                else { bgm.volume = 0; bgm.pause(); clearInterval(fadeOut); }
            }, 50);

            setTimeout(() => { document.getElementById('redirect-link').style.display = 'block'; }, 4000);
            setTimeout(() => { window.location.href = TARGET_URL; }, 5000); 
        }

        btn.addEventListener('mousedown', clickHandler);
        btn.addEventListener('touchstart', clickHandler);
        startDecay();

    </script>
